// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

import "hardhat/console.sol";

contract WavePortal {
    uint256 totalWaves;
    
    // This is a "struct" - a custom data type
    struct Wave {
        address sender; // The address of the person who waved
        string message;  // The message they sent
        uint256 timestamp; // The time the wave happened
    }

    // This is an array that will store all the Wave structs
    Wave[] waves;

    // This is an "event" - it's like a broadcast.
    // Our web app can listen for this to know a new wave happened.
    event NewWave(address indexed sender, string message, uint256 timestamp);

    constructor() {
        console.log("I am a smart contract, and I am ALIVE!");
    }

    // This function now takes a '_message' string
    function wave(string memory _message) public {
        totalWaves += 1;
        console.log("%s has waved with message: %s", msg.sender, _message);

        // Add a new Wave struct to our 'waves' array
        waves.push(Wave(msg.sender, _message, block.timestamp));

        // Emit the event so our web app can see it
        emit NewWave(msg.sender, _message, block.timestamp);
    }

    // Gets the total wave count
    function getTotalWaves() public view returns (uint256) {
        return totalWaves;
    }

    // Gets the full array of all Wave structs
    function getAllWaves() public view returns (Wave[] memory) {
        return waves;
    }
}

Step 3: Deploy to the Sepolia Testnet

- update deploy.js

const hre = require("hardhat");

async function main() {
  // Get the "recipe" for the contract
  const waveContractFactory = await hre.ethers.getContractFactory("WavePortal");
  
  // "Deploy" the contract (build it and put it on the blockchain)
  console.log("Deploying contract to Sepolia...");
  const waveContract = await waveContractFactory.deploy();
  
  // Wait for it to be officially deployed
  await waveContract.waitForDeployment();
  
  // Log the final address
  const contractAddress = await waveContract.getAddress();
  console.log("Contract deployed to:", contractAddress);
}

// Standard pattern to run the 'main' function and catch errors
main()
  .then(() => process.exit(0))
  .catch((error) => {
    console.error(error);
    process.exit(1);
  });
